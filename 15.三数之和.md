/*并不满足题解的代码，返回的三元组之中可以有重复的数字，但是以下代码不满足*/
int compare (const void * a,const  void * b)
{
 return ( *(int*)a - *(int*)b );
}
int** threeSum(int* nums, int numsSize, int* returnSize, int** returnColumnSizes){

*returnSize = 0;
int **ret = (int **)malloc(sizeof(int *) * numsSize  *numsSize);
*returnColumnSizes = (int *)malloc(sizeof(int) * numsSize * numsSize);
// 排序为升序
qsort(nums,numsSize,sizeof(int),compare);
int length = numsSize;
int * sets = (int*)malloc(sizeof(int) * length);  
memset(sets, 0, sizeof(int) * length);       
// 填充首元素
sets[0] = nums[0];
int idx = 1; 

// 循环建立哈希表
for(int i = 1;i < length;i++)
	if(nums[i] > sets[idx-1])
		sets[idx++] = nums[i];

for(int j = 0;j <= idx-3 ;j++)
    if(sets[j]+sets[j+1]+sets[j+2] == 0)
        {
        ret[*returnSize] = (int*)malloc(sizeof(int)*3);
        (*returnColumnSizes)[*returnSize] = 3;
        ret[*returnSize][0] = sets[j];
        ret[*returnSize][1] = sets[j+1];
        ret[*returnSize][2] = sets[j+2];
        (*returnSize)++;
        }
    return ret;
}
