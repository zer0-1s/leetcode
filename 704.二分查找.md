}

int search(int* nums, int numsSize, int target){
    // 定义左闭右闭区间
    int min = 0,max = numsSize-1;
    // int mid = (min + max)/2;
    while(min <= max)
    {   // 防止溢出,可能存在
        int mid = min + (max-min)/2;
        if(nums[mid] < target)
            min = mid + 1;
        else if(nums[mid] > target)
            max = mid - 1;
        else
            return mid;
    }
    // 找不到目标
    return -1;
    
}
