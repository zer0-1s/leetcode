bool isAnagram(char * s, char * t){
    // set hashmap1
    int hashmap1[26] = {0};
    // set hashmap2
    int hashmap2[26] = {0};
    if(strlen(s) == strlen(t))
    {
        int flag = 0; 
    for(int i = 0;i < strlen(s); i++)
        hashmap1[s[i]%97]+=1;
    for(int j = 0;j < strlen(t); j++)
        hashmap2[t[j]%97]+=1;
    for(int j = 0;j < strlen(t); j++)
       { 
        if(hashmap2[t[j]%97] == hashmap1[t[j]%97]) 
         flag++; 
        }
        if(flag == strlen(t))
            return true;
    } 
    
    return false;
}
